<script>
const form = document.getElementById("mangoForm");

form.addEventListener("submit", function(e) {
  e.preventDefault();

  fetch("PASTE_YOUR_URL_HERE", {
    method: "POST",
    body: new FormData(form)
  })
  .then(res => res.text())
  .then(data => {
    alert("Order placed successfully 🍋");
    form.reset();
  })
  .catch(err => {
    alert("Error! Try again");
  });
});
</script>
